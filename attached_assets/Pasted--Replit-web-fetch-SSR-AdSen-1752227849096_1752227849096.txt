🎯 구체적 수정 요청사항:
1. web_fetch 도구를 봇으로 인식하도록 설정
javascript// 봇 감지 로직에 추가 필요한 패턴들:

  // ... 기존 패턴들
  'claude', 'anthropic', 'web_fetch', 'fetch', 'crawler',
  'python-requests', 'curl', 'wget', 'http-client'
];
2. 모든 의심스러운 요청에 SSR 제공
javascript// 더 포괄적인 봇 감지 또는 기본값을 SSR로 설정
const shouldProvideSSR = (userAgent) => {
  // 브라우저가 아닌 모든 요청에 SSR 제공
  return !isBrowser(userAgent) || isBot(userAgent);
};
3. 완전한 SSR 콘텐츠 보장
javascript// 반드시 포함되어야 할 콘텐츠:
